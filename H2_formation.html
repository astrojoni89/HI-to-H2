def transition(alpha_G) : #with sigma_g_rel approx. 1
    return 6.71 * np.log((alpha_G/3.2)+1)

def alpha_G(J_FUV, n) : #with sigma_g_rel approx. 1
    return 2.0 * J_FUV * (30/n) * (9.9/1+8.9)**0.37
